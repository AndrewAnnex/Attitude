def __simple_ellipse(**kwargs):
    for i in range(10):
        obj = random_pca()
        args = (obj.axes, obj.covariance_matrix)
        v1 = normal_errors(*args, **kwargs)
        v2 = iterative_normal_errors(*args, **kwargs)
        assert len(v1) == len(v2)
        assert N.allclose(v1,v2)
