[docs]def encode(fig):
    b = BytesIO()
    fig.savefig(b, format='png',bbox_inches='tight')
    b.seek(0)
    return '<img src ="{},{}"/>'.format(
            'data:image/png;base64',
            quote(b64encode(b.read())))
